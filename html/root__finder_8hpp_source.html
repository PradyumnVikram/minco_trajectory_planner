42 inline int polyMod(double *u, double *v, double *r, int lu, int lv)
43 // Modulus of u(x)/v(x)
44 // The leading coefficient of v, i.e., v[0], must be 1.0 or -1.0
45 // The length of u, v, and r are lu, lv, and lu, respectively
46 {
47  int orderu = lu - 1;
48  int orderv = lv - 1;
49 
50  memcpy(r, u, lu * sizeof(double));
51 
52  if (v[0] < 0.0)
53  {
54  for (int i = orderv + 1; i <= orderu; i += 2)
55  {
56  r[i] = -r[i];
57  }
58  for (int i = 0; i <= orderu - orderv; i++)
59  {
60  for (int j = i + 1; j <= orderv + i; j++)
61  {
62  r[j] = -r[j] - r[i] * v[j - i];
63  }
64  }
65  }
66  else
67  {
68  for (int i = 0; i <= orderu - orderv; i++)
69  {
70  for (int j = i + 1; j <= orderv + i; j++)
71  {
72  r[j] = r[j] - r[i] * v[j - i];
73  }
74  }
75  }
76 
77  int k = orderv - 1;
78  while (k >= 0 && fabs(r[orderu - k]) < DBL_EPSILON)
79  {
80  r[orderu - k] = 0.0;
81  k--;
82  }
83 
84  return (k <= 0) ? 1 : (k + 1);
85 }

87 inline double polyEval(double *p, int len, double x)
88 // Evaluate the polynomial p(x), which has len coefficients
89 // Note: Horner scheme should not be employed here !!!
90 // Horner scheme has bad numerical stability despite of its efficiency.
91 // These errors are particularly troublesome for root-finding algorithms.
92 // When the polynomial is evaluated near a zero, catastrophic
93 // cancellation (subtracting two nearby numbers) is guaranteed to occur.
94 // Therefore, Horner scheme may slow down some root-finding algorithms.
95 {
96  double retVal = 0.0;
97 
98  if (len > 0)
99  {
100  if (fabs(x) < DBL_EPSILON)
101  {
102  retVal = p[len - 1];
103  }
104  else if (x == 1.0)
105  {
106  for (int i = len - 1; i >= 0; i--)
107  {
108  retVal += p[i];
109  }
110  }
111  else
112  {
113  double xn = 1.0;
114 
115  for (int i = len - 1; i >= 0; i--)
116  {
117  retVal += p[i] * xn;
118  xn *= x;
119  }
120  }
121  }
122 
123  return retVal;
124 }

126 inline std::set<double> solveCub(double a, double b, double c, double d)
127 // Calculate all roots of a*x^3 + b*x^2 + c*x + d = 0
128 {
129  std::set<double> roots;
130 
131  constexpr double cos120 = -0.50;
132  constexpr double sin120 = 0.866025403784438646764;
133 
134  if (fabs(d) < DBL_EPSILON)
135  {
136  // First solution is x = 0
137  roots.insert(0.0);
138 
139  // Converting to a quadratic equation
140  d = c;
141  c = b;
142  b = a;
143  a = 0.0;
144  }
145 
146  if (fabs(a) < DBL_EPSILON)
147  {
148  if (fabs(b) < DBL_EPSILON)
149  {
150  // Linear equation
151  if (fabs(c) > DBL_EPSILON)
152  roots.insert(-d / c);
153  }
154  else
155  {
156  // Quadratic equation
157  double discriminant = c * c - 4.0 * b * d;
158  if (discriminant >= 0)
159  {
160  double inv2b = 1.0 / (2.0 * b);
161  double y = sqrt(discriminant);
162  roots.insert((-c + y) * inv2b);
163  roots.insert((-c - y) * inv2b);
164  }
165  }
166  }
167  else
168  {
169  // Cubic equation
170  double inva = 1.0 / a;
171  double invaa = inva * inva;
172  double bb = b * b;
173  double bover3a = b * (1.0 / 3.0) * inva;
174  double p = (3.0 * a * c - bb) * (1.0 / 3.0) * invaa;
175  double halfq = (2.0 * bb * b - 9.0 * a * b * c + 27.0 * a * a * d) * (0.5 / 27.0) * invaa * inva;
176  double yy = p * p * p / 27.0 + halfq * halfq;
177 
178  if (yy > DBL_EPSILON)
179  {
180  // Sqrt is positive: one real solution
181  double y = sqrt(yy);
182  double uuu = -halfq + y;
183  double vvv = -halfq - y;
184  double www = fabs(uuu) > fabs(vvv) ? uuu : vvv;
185  double w = (www < 0) ? -pow(fabs(www), 1.0 / 3.0) : pow(www, 1.0 / 3.0);
186  roots.insert(w - p / (3.0 * w) - bover3a);
187  }
188  else if (yy < -DBL_EPSILON)
189  {
190  // Sqrt is negative: three real solutions
191  double x = -halfq;
192  double y = sqrt(-yy);
193  double theta;
194  double r;
195  double ux;
196  double uyi;
197  // Convert to polar form
198  if (fabs(x) > DBL_EPSILON)
199  {
200  theta = (x > 0.0) ? atan(y / x) : (atan(y / x) + M_PI);
201  r = sqrt(x * x - yy);
202  }
203  else
204  {
205  // Vertical line
206  theta = M_PI / 2.0;
207  r = y;
208  }
209  // Calculate cube root
210  theta /= 3.0;
211  r = pow(r, 1.0 / 3.0);
212  // Convert to complex coordinate
213  ux = cos(theta) * r;
214  uyi = sin(theta) * r;
215  // First solution
216  roots.insert(ux + ux - bover3a);
217  // Second solution, rotate +120 degrees
218  roots.insert(2.0 * (ux * cos120 - uyi * sin120) - bover3a);
219  // Third solution, rotate -120 degrees
220  roots.insert(2.0 * (ux * cos120 + uyi * sin120) - bover3a);
221  }
222  else
223  {
224  // Sqrt is zero: two real solutions
225  double www = -halfq;
226  double w = (www < 0.0) ? -pow(fabs(www), 1.0 / 3.0) : pow(www, 1.0 / 3.0);
227  // First solution
228  roots.insert(w + w - bover3a);
229  // Second solution, rotate +120 degrees
230  roots.insert(2.0 * w * cos120 - bover3a);
231  }
232  }
233  return roots;
234 }

236 inline int solveResolvent(double *x, double a, double b, double c)
237 // Solve resolvent eqaution of corresponding Quartic equation
238 // The input x must be of length 3
239 // Number of zeros are returned
240 {
241  double a2 = a * a;
242  double q = (a2 - 3.0 * b) / 9.0;
243  double r = (a * (2.0 * a2 - 9.0 * b) + 27.0 * c) / 54.0;
244  double r2 = r * r;
245  double q3 = q * q * q;
246  double A, B;
247  if (r2 < q3)
248  {
249  double t = r / sqrt(q3);
250  if (t < -1.0)
251  {
252  t = -1.0;
253  }
254  if (t > 1.0)
255  {
256  t = 1.0;
257  }
258  t = acos(t);
259  a /= 3.0;
260  q = -2.0 * sqrt(q);
261  x[0] = q * cos(t / 3.0) - a;
262  x[1] = q * cos((t + M_PI * 2.0) / 3.0) - a;
263  x[2] = q * cos((t - M_PI * 2.0) / 3.0) - a;
264  return 3;
265  }
266  else
267  {
268  A = -pow(fabs(r) + sqrt(r2 - q3), 1.0 / 3.0);
269  if (r < 0.0)
270  {
271  A = -A;
272  }
273  B = (0.0 == A ? 0.0 : q / A);
274 
275  a /= 3.0;
276  x[0] = (A + B) - a;
277  x[1] = -0.5 * (A + B) - a;
278  x[2] = 0.5 * sqrt(3.0) * (A - B);
279  if (fabs(x[2]) < DBL_EPSILON)
280  {
281  x[2] = x[1];
282  return 2;
283  }
284 
285  return 1;
286  }

289 inline std::set<double> solveQuartMonic(double a, double b, double c, double d)
290 // Calculate all roots of the monic quartic equation:
291 // x^4 + a*x^3 + b*x^2 + c*x +d = 0
292 {
293  std::set<double> roots;
294 
295  double a3 = -b;
296  double b3 = a * c - 4.0 * d;
297  double c3 = -a * a * d - c * c + 4.0 * b * d;
298 
299  // Solve the resolvent: y^3 - b*y^2 + (ac - 4*d)*y - a^2*d - c^2 + 4*b*d = 0
300  double x3[3];
301  int iZeroes = solveResolvent(x3, a3, b3, c3);
302 
303  double q1, q2, p1, p2, D, sqrtD, y;
304 
305  y = x3[0];
306  // Choosing Y with maximal absolute value.
307  if (iZeroes != 1)
308  {
309  if (fabs(x3[1]) > fabs(y))
310  {
311  y = x3[1];
312  }
313  if (fabs(x3[2]) > fabs(y))
314  {
315  y = x3[2];
316  }
317  }
318 
319  // h1 + h2 = y && h1*h2 = d <=> h^2 - y*h + d = 0 (h === q)
320 
321  D = y * y - 4.0 * d;
322  if (fabs(D) < DBL_EPSILON) //In other words: D == 0
323  {
324  q1 = q2 = y * 0.5;
325  // g1 + g2 = a && g1 + g2 = b - y <=> g^2 - a*g + b - y = 0 (p === g)
326  D = a * a - 4.0 * (b - y);
327  if (fabs(D) < DBL_EPSILON) //In other words: D == 0
328  {
329  p1 = p2 = a * 0.5;
330  }
331  else
332  {
333  sqrtD = sqrt(D);
334  p1 = (a + sqrtD) * 0.5;
335  p2 = (a - sqrtD) * 0.5;
336  }
337  }
338  else
339  {
340  sqrtD = sqrt(D);
341  q1 = (y + sqrtD) * 0.5;
342  q2 = (y - sqrtD) * 0.5;
343  // g1 + g2 = a && g1*h2 + g2*h1 = c ( && g === p ) Krammer
344  p1 = (a * q1 - c) / (q1 - q2);
345  p2 = (c - a * q2) / (q1 - q2);
346  }
347 
348  // Solve the quadratic equation: x^2 + p1*x + q1 = 0
349  D = p1 * p1 - 4.0 * q1;
350  if (fabs(D) < DBL_EPSILON)
351  {
352  roots.insert(-p1 * 0.5);
353  }
354  else if (D > 0.0)
355  {
356  sqrtD = sqrt(D);
357  roots.insert((-p1 + sqrtD) * 0.5);
358  roots.insert((-p1 - sqrtD) * 0.5);
359  }
360 
361  // Solve the quadratic equation: x^2 + p2*x + q2 = 0
362  D = p2 * p2 - 4.0 * q2;
363  if (fabs(D) < DBL_EPSILON)
364  {
365  roots.insert(-p2 * 0.5);
366  }
367  else if (D > 0.0)
368  {
369  sqrtD = sqrt(D);
370  roots.insert((-p2 + sqrtD) * 0.5);
371  roots.insert((-p2 - sqrtD) * 0.5);
372  }
373 
374  return roots;
375 }

377 inline std::set<double> solveQuart(double a, double b, double c, double d, double e)
378 // Calculate the quartic equation: a*x^4 + b*x^3 + c*x^2 + d*x + e = 0
379 // All coefficients can be zero
380 {
381  if (fabs(a) < DBL_EPSILON)
382  {
383  return solveCub(b, c, d, e);
384  }
385  else
386  {
387  return solveQuartMonic(b / a, c / a, d / a, e / a);
388  }

391 inline std::set<double> eigenSolveRealRoots(const Eigen::VectorXd &coeffs, double lbound, double ubound, double tol)
392 // Calculate roots of coeffs(x) inside (lbound, rbound) by computing eigen values of its companion matrix
393 // Complex roots with magnitude of imaginary part less than tol are considered real
394 {
395  std::set<double> rts;
396 
397  int order = (int)coeffs.size() - 1;
398  Eigen::VectorXd monicCoeffs(order + 1);
399  monicCoeffs << 1.0, coeffs.tail(order) / coeffs(0);
400 
401  Eigen::MatrixXd companionMat(order, order);
402  companionMat.setZero();
403  companionMat(0, order - 1) = -monicCoeffs(order);
404  for (int i = 1; i < order; i++)
405  {
406  companionMat(i, i - 1) = 1.0;
407  companionMat(i, order - 1) = -monicCoeffs(order - i);
408  }
409  Eigen::VectorXcd eivals = companionMat.eigenvalues();
410  double real;
411  int eivalsNum = eivals.size();
412  for (int i = 0; i < eivalsNum; i++)
413  {
414  real = eivals(i).real();
415  if (eivals(i).imag() < tol && real > lbound && real < ubound)
416  rts.insert(real);
417  }
418 
419  return rts;
420 }

422 inline double numSignVar(double x, double **sturmSeqs, int *szSeq, int len)
423 // Calculate the number of sign variations of the Sturm sequences at x
424 // The i-th sequence with size szSeq[i] stored in sturmSeqs[i][], 0 <= i < len
425 {
426  double y, lasty;
427  int signVar = 0;
428  lasty = polyEval(sturmSeqs[0], szSeq[0], x);
429  for (int i = 1; i < len; i++)
430  {
431  y = polyEval(sturmSeqs[i], szSeq[i], x);
432  if (lasty == 0.0 || lasty * y < 0.0)
433  {
434  ++signVar;
435  }
436  lasty = y;
437  }
438 
439  return signVar;

442 inline void polyDeri(double *coeffs, double *dcoeffs, int len)
443 // Calculate the derivative poly coefficients of a given poly
444 {
445  int horder = len - 1;
446  for (int i = 0; i < horder; i++)
447  {
448  dcoeffs[i] = (horder - i) * coeffs[i];
449  }
450  return;

453 template <typename F, typename DF>
454 inline double safeNewton(const F &func, const DF &dfunc,
455  const double &l, const double &h,
456  const double &tol, const int &maxIts)
457 // Safe Newton Method
458 // Requirements: f(l)*f(h)<=0
459 {
460  double xh, xl;
461  double fl = func(l);
462  double fh = func(h);
463  if (fl == 0.0)
464  {
465  return l;
466  }
467  if (fh == 0.0)
468  {
469  return h;
470  }
471  if (fl < 0.0)
472  {
473  xl = l;
474  xh = h;
475  }
476  else
477  {
478  xh = l;
479  xl = h;
480  }
481 
482  double rts = 0.5 * (xl + xh);
483  double dxold = fabs(xh - xl);
484  double dx = dxold;
485  double f = func(rts);
486  double df = dfunc(rts);
487  double temp;
488  for (int j = 0; j < maxIts; j++)
489  {
490  if ((((rts - xh) * df - f) * ((rts - xl) * df - f) > 0.0) ||
491  (fabs(2.0 * f) > fabs(dxold * df)))
492  {
493  dxold = dx;
494  dx = 0.5 * (xh - xl);
495  rts = xl + dx;
496  if (xl == rts)
497  {
498  break;
499  }
500  }
501  else
502  {
503  dxold = dx;
504  dx = f / df;
505  temp = rts;
506  rts -= dx;
507  if (temp == rts)
508  {
509  break;
510  }
511  }
512 
513  if (fabs(dx) < tol)
514  {
515  break;
516  }
517 
518  f = func(rts);
519  df = dfunc(rts);
520  if (f < 0.0)
521  {
522  xl = rts;
523  }
524  else
525  {
526  xh = rts;
527  }
528  }
529 
530  return rts;
531 }

533 inline double shrinkInterval(double *coeffs, int numCoeffs, double lbound, double ubound, double tol)
534 // Calculate a single zero of poly coeffs(x) inside [lbound, ubound]
535 // Requirements: coeffs(lbound)*coeffs(ubound) < 0, lbound < ubound
536 {
537  double *dcoeffs = new double[numCoeffs - 1];
538  polyDeri(coeffs, dcoeffs, numCoeffs);
539  auto func = [&coeffs, &numCoeffs](double x) { return polyEval(coeffs, numCoeffs, x); };
540  auto dfunc = [&dcoeffs, &numCoeffs](double x) { return polyEval(dcoeffs, numCoeffs - 1, x); };
541  constexpr int maxDblIts = 128;
542  double rts = safeNewton(func, dfunc, lbound, ubound, tol, maxDblIts);
543  delete[] dcoeffs;
544  return rts;
545 }

547 inline void recurIsolate(double l, double r, double fl, double fr, int lnv, int rnv,
548  double tol, double **sturmSeqs, int *szSeq, int len,
549  std::set<double> &rts)
550 // Isolate all roots of sturmSeqs[0](x) inside interval (l, r) recursively and store them in rts
551 // Requirements: fl := sturmSeqs[0](l) != 0, fr := sturmSeqs[0](r) != 0, l < r,
552 // lnv != rnv, lnv = numSignVar(l), rnv = numSignVar(r)
553 // sturmSeqs[0](x) must have at least one root inside (l, r)
554 {
555  int nrts = lnv - rnv;
556  double fm;
557  double m;
558 
559  if (nrts == 0)
560  {
561  return;
562  }
563  else if (nrts == 1)
564  {
565  if (fl * fr < 0)
566  {
567  rts.insert(shrinkInterval(sturmSeqs[0], szSeq[0], l, r, tol));
568  return;
569  }
570  else
571  {
572  // Bisect when non of above works
573  int maxDblIts = 128;
574 
575  for (int i = 0; i < maxDblIts; i++)
576  {
577  // Calculate the root with even multiplicity
578  if (fl * fr < 0)
579  {
580  rts.insert(shrinkInterval(sturmSeqs[1], szSeq[1], l, r, tol));
581  return;
582  }
583 
584  m = (l + r) / 2.0;
585  fm = polyEval(sturmSeqs[0], szSeq[0], m);
586 
587  if (fm == 0 || fabs(r - l) < tol)
588  {
589  rts.insert(m);
590  return;
591  }
592  else
593  {
594  if (lnv == numSignVar(m, sturmSeqs, szSeq, len))
595  {
596  l = m;
597  fl = fm;
598  }
599  else
600  {
601  r = m;
602  fr = fm;
603  }
604  }
605  }
606 
607  rts.insert(m);
608  return;
609  }
610  }
611  else if (nrts > 1)
612  {
613  // More than one root exists in the interval
614  int maxDblIts = 128;
615 
616  int mnv;
617  int bias = 0;
618  bool biased = false;
619  for (int i = 0; i < maxDblIts; i++)
620  {
621  bias = biased ? bias : 0;
622  if (!biased)
623  {
624  m = (l + r) / 2.0;
625  }
626  else
627  {
628  m = (r - l) / pow(2.0, bias + 1.0) + l;
629  biased = false;
630  }
631  mnv = numSignVar(m, sturmSeqs, szSeq, len);
632 
633  if (fabs(r - l) < tol)
634  {
635  rts.insert(m);
636  return;
637  }
638  else
639  {
640  fm = polyEval(sturmSeqs[0], szSeq[0], m);
641  if (fm == 0)
642  {
643  bias++;
644  biased = true;
645  }
646  else if (lnv != mnv && rnv != mnv)
647  {
648  recurIsolate(l, m, fl, fm, lnv, mnv, tol, sturmSeqs, szSeq, len, rts);
649  recurIsolate(m, r, fm, fr, mnv, rnv, tol, sturmSeqs, szSeq, len, rts);
650  return;
651  }
652  else if (lnv == mnv)
653  {
654  l = m;
655  fl = fm;
656  }
657  else
658  {
659  r = m;
660  fr = fm;
661  }
662  }
663  }
664 
665  rts.insert(m);
666  return;
667  }

670 inline std::set<double> isolateRealRoots(const Eigen::VectorXd &coeffs, double lbound, double ubound, double tol)
671 // Calculate roots of coeffs(x) inside (lbound, rbound) leveraging Sturm theory
672 // Requirement: leading coefficient must be nonzero
673 // coeffs(lbound) != 0, coeffs(rbound) != 0, lbound < rbound
674 {
675  std::set<double> rts;
676 
677  // Calculate monic coefficients
678  int order = (int)coeffs.size() - 1;
679  Eigen::VectorXd monicCoeffs(order + 1);
680  monicCoeffs << 1.0, coeffs.tail(order) / coeffs(0);
681 
682  // Calculate Cauchys bound for the roots of a polynomial
683  double rho_c = 1 + monicCoeffs.tail(order).cwiseAbs().maxCoeff();
684 
685  // Calculate Kojimas bound for the roots of a polynomial
686  Eigen::VectorXd nonzeroCoeffs(order + 1);
687  nonzeroCoeffs.setZero();
688  int nonzeros = 0;
689  double tempEle;
690  for (int i = 0; i < order + 1; i++)
691  {
692  tempEle = monicCoeffs(i);
693  if (fabs(tempEle) >= DBL_EPSILON)
694  {
695  nonzeroCoeffs(nonzeros++) = tempEle;
696  }
697  }
698  nonzeroCoeffs = nonzeroCoeffs.head(nonzeros).eval();
699  Eigen::VectorXd kojimaVec = nonzeroCoeffs.tail(nonzeros - 1).cwiseQuotient(nonzeroCoeffs.head(nonzeros - 1)).cwiseAbs();
700  kojimaVec.tail(1) /= 2.0;
701  double rho_k = 2.0 * kojimaVec.maxCoeff();
702 
703  // Choose a sharper one then loosen it by 1.0 to get an open interval
704  double rho = std::min(rho_c, rho_k) + 1.0;
705 
706  // Tighten the bound to search in
707  lbound = std::max(lbound, -rho);
708  ubound = std::min(ubound, rho);
709 
710  // Build Sturm sequence
711  int len = monicCoeffs.size();
712  double sturmSeqs[(RootFinderParam::highestOrder + 1) * (RootFinderParam::highestOrder + 1)];
713  int szSeq[RootFinderParam::highestOrder + 1] = {0}; // Explicit ini as zero (gcc may neglect this in -O3)
714  double *offsetSeq[RootFinderParam::highestOrder + 1];
715  int num = 0;
716 
717  for (int i = 0; i < len; i++)
718  {
719  sturmSeqs[i] = monicCoeffs(i);
720  sturmSeqs[i + 1 + len] = (order - i) * sturmSeqs[i] / order;
721  }
722  szSeq[0] = len;
723  szSeq[1] = len - 1;
724  offsetSeq[0] = sturmSeqs + len - szSeq[0];
725  offsetSeq[1] = sturmSeqs + 2 * len - szSeq[1];
726 
727  num += 2;
728 
729  bool remainderConstant = false;
730  int idx = 0;
731  while (!remainderConstant)
732  {
733  szSeq[idx + 2] = polyMod(offsetSeq[idx],
734  offsetSeq[idx + 1],
735  &(sturmSeqs[(idx + 3) * len - szSeq[idx]]),
736  szSeq[idx], szSeq[idx + 1]);
737  offsetSeq[idx + 2] = sturmSeqs + (idx + 3) * len - szSeq[idx + 2];
738 
739  remainderConstant = szSeq[idx + 2] == 1;
740  for (int i = 1; i < szSeq[idx + 2]; i++)
741  {
742  offsetSeq[idx + 2][i] /= -fabs(offsetSeq[idx + 2][0]);
743  }
744  offsetSeq[idx + 2][0] = offsetSeq[idx + 2][0] > 0.0 ? -1.0 : 1.0;
745  num++;
746  idx++;
747  }
748 
749  // Isolate all distinct roots inside the open interval recursively
750  recurIsolate(lbound, ubound,
751  polyEval(offsetSeq[0], szSeq[0], lbound),
752  polyEval(offsetSeq[0], szSeq[0], ubound),
753  numSignVar(lbound, offsetSeq, szSeq, len),
754  numSignVar(ubound, offsetSeq, szSeq, len),
755  tol, offsetSeq, szSeq, len, rts);
756 
757  return rts;
758 }

765 inline Eigen::VectorXd polyConv(const Eigen::VectorXd &lCoef, const Eigen::VectorXd &rCoef)
766 // Calculate the convolution of lCoef(x) and rCoef(x)
767 {
768  Eigen::VectorXd result(lCoef.size() + rCoef.size() - 1);
769  result.setZero();

771  {
772  for (int j = 0; j <= i; j++)

850 inline Eigen::VectorXd polySqr(const Eigen::VectorXd &coef)
851 // Calculate self-convolution of coef(x)
852 {
853  int coefSize = coef.size();
854  int resultSize = coefSize * 2 - 1;
855  int lbound, rbound;
856  Eigen::VectorXd result(resultSize);
857  double temp;
858  for (int i = 0; i < resultSize; i++)
859  {
860  temp = 0;
861  lbound = i - coefSize + 1;
862  lbound = lbound > 0 ? lbound : 0;
863  rbound = coefSize < (i + 1) ? coefSize : (i + 1);
864  rbound += lbound;
865  if (rbound & 1) //faster than rbound % 2 == 1
866  {
867  rbound >>= 1; //faster than rbound /= 2
868  temp += coef(rbound) * coef(rbound);
869  }
870  else
871  {
872  rbound >>= 1; //faster than rbound /= 2
873  }
874 
875  for (int j = lbound; j < rbound; j++)
876  {
877  temp += 2.0 * coef(j) * coef(i - j);
878  }
879  result(i) = temp;
880  }
881 
882  return result;
883 }

885 inline double polyVal(const Eigen::VectorXd &coeffs, double x,
886  bool numericalStability = true)
887 // Evaluate the polynomial at x, i.e., coeffs(x)
888 // Horner scheme is faster yet less stable
889 // Stable one should be used when coeffs(x) is close to 0.0
890 {

893

895  {

898  retVal = coeffs(order);

931 inline int countRoots(const Eigen::VectorXd &coeffs, double l, double r)
932 // Count the number of distinct roots of coeffs(x) inside (l, r), leveraging Sturm theory
933 // Boundary values, i.e., coeffs(l) and coeffs(r), must be nonzero
934 {
935  int nRoots = 0;
936 
937  int originalSize = coeffs.size();
938  int valid = originalSize;
939  for (int i = 0; i < originalSize; i++)
940  {
941  if (fabs(coeffs(i)) < DBL_EPSILON)
942  {
943  valid--;
944  }
945  else
946  {
947  break;
948  }
949  }
950 
951  if (valid > 0 && fabs(coeffs(originalSize - 1)) > DBL_EPSILON)
952  {
953  Eigen::VectorXd monicCoeffs(valid);
954  monicCoeffs << 1.0, coeffs.segment(originalSize - valid + 1, valid - 1) / coeffs(originalSize - valid);
955 
956  // Build the Sturm sequence
957  int len = monicCoeffs.size();
958  int order = len - 1;
959  double sturmSeqs[(RootFinderParam::highestOrder + 1) * (RootFinderParam::highestOrder + 1)];
960  int szSeq[RootFinderParam::highestOrder + 1] = {0}; // Explicit ini as zero (gcc may neglect this in -O3)
961  int num = 0;
962 
963  for (int i = 0; i < len; i++)
964  {
965  sturmSeqs[i] = monicCoeffs(i);
966  sturmSeqs[i + 1 + len] = (order - i) * sturmSeqs[i] / order;
967  }
968  szSeq[0] = len;
969  szSeq[1] = len - 1;
970  num += 2;
971 
972  bool remainderConstant = false;
973  int idx = 0;
974  while (!remainderConstant)
975  {
976  szSeq[idx + 2] = RootFinderPriv::polyMod(&(sturmSeqs[(idx + 1) * len - szSeq[idx]]),
977  &(sturmSeqs[(idx + 2) * len - szSeq[idx + 1]]),
978  &(sturmSeqs[(idx + 3) * len - szSeq[idx]]),
979  szSeq[idx], szSeq[idx + 1]);
980  remainderConstant = szSeq[idx + 2] == 1;
981  for (int i = 1; i < szSeq[idx + 2]; i++)
982  {
983  sturmSeqs[(idx + 3) * len - szSeq[idx + 2] + i] /= -fabs(sturmSeqs[(idx + 3) * len - szSeq[idx + 2]]);
984  }
985  sturmSeqs[(idx + 3) * len - szSeq[idx + 2]] /= -fabs(sturmSeqs[(idx + 3) * len - szSeq[idx + 2]]);
986  num++;
987  idx++;
988  }
989 
990  // Count numbers of sign variations at two boundaries
991  double yl, lastyl, yr, lastyr;
992  lastyl = RootFinderPriv::polyEval(&(sturmSeqs[len - szSeq[0]]), szSeq[0], l);
993  lastyr = RootFinderPriv::polyEval(&(sturmSeqs[len - szSeq[0]]), szSeq[0], r);
994  for (int i = 1; i < num; i++)
995  {
996  yl = RootFinderPriv::polyEval(&(sturmSeqs[(i + 1) * len - szSeq[i]]), szSeq[i], l);
997  yr = RootFinderPriv::polyEval(&(sturmSeqs[(i + 1) * len - szSeq[i]]), szSeq[i], r);
998  if (lastyl == 0.0 || lastyl * yl < 0.0)
999  {
1000  ++nRoots;
1001  }
1002  if (lastyr == 0.0 || lastyr * yr < 0.0)
1003  {
1004  --nRoots;
1005  }
1006  lastyl = yl;
1007  lastyr = yr;
1008  }
1009  }
1010 
1011  return nRoots;
