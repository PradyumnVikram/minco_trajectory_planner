521  const Vector3<T> &b,
522  const Vector3<T> &c)
523  {

525  // without constructing temp vectors
526  T x = a.x - c.x;

528  T z = a.z - c.z;

534  T pz = x * rhsy - y * rhsx;

588  }
589  }

591  }
592 
593  i = 0;
594  for (const auto &halfEdge : builderObject.m_halfEdges)
595  {
596  if (!halfEdge.isDisabled())

598  m_halfEdges.push_back({static_cast<IndexType>(halfEdge.m_endVertex),
599  static_cast<IndexType>(halfEdge.m_opp),

611  }

614  {
615  he.m_face = faceMapping[he.m_face];
616  he.m_opp = halfEdgeMapping[he.m_opp];
617  he.m_next = halfEdgeMapping[he.m_next];

717  std::vector<size_t> faceStack;
718  // Map vertex indices from original point cloud

721  for (size_t i = 0; i < mesh.m_faces.size(); i++)
722  {

841  // Implementation of the 3D QuickHull algorithm by Antti Kuukka
842  //
843  // No copyrights. What follows is 100% Public Domain.
844  //
845  //
846  //
847  // INPUT: a list of points in 3D space (for example, vertices of a 3D mesh)
848  //
849  // OUTPUT: a ConvexHull object which provides vertex and index buffers of
850  // the generated convex hull as a triangle mesh.
851  //
852  //
853  //
854  // The implementation is thread-safe if each thread is using its own
855  // QuickHull object.
856  //
857  //
858  // SUMMARY OF THE ALGORITHM:
859  // - Create initial simplex (tetrahedron) using extreme points.
860  // We have four faces now and they form a convex mesh M.
861  // - For each point, assign them to the first face for which they
862  // are on the positive side of (so each point is assigned to at
863  // most one face). Points inside the initial tetrahedron are left
864  // behind now and no longer affect the calculations.
865  // - Add all faces that have points assigned to them to Face Stack.
866  // - Iterate until Face Stack is empty:
867  // - Pop topmost face F from the stack
868  // - From the points assigned to F, pick the point P that is
869  // farthest away from the plane defined by F.
870  // - Find all faces of M that have P on their positive side.
871  // Let us call these the "visible faces".
872  // - Because of the way M is constructed, these faces are connected.
873  // Solve their horizon edge loop.
874  // - "Extrude to P": Create new faces by connecting P with the points
875  // belonging to the horizon edge. Add the new faces to M and remove
876  // the visible faces from M.
877  // - Each point that was assigned to visible faces is now assigned
878  // to at most one of the newly created faces.
879  // - Those new faces that have points assigned to them are added
880  // to the top of Face Stack.
881  // - M is now the convex hull.
882  //
883  // TO DO:
884  // - Implement a proper 2D QuickHull and use that to solve the degenerate 2D case
885  // (when all the points lie on the same plane in 3D space).

931  // Create a half edge mesh representing the base tetrahedron from which the QuickHull
932  // iteration proceeds. m_extremeValues must be properly set up when this is called.
933  inline void setupInitialTetrahedron()
934  {
935  const size_t vertexCount = m_vertexData.size();
936 
937  // If we have at most 4 points, just return a degenerate tetrahedron:
938  if (vertexCount <= 4)
939  {
940  size_t v[4] = {0,
941  std::min((size_t)1, vertexCount - 1),
942  std::min((size_t)2, vertexCount - 1),
943  std::min((size_t)3, vertexCount - 1)};
944  const Vector3<T> N = mathutils::getTriangleNormal(m_vertexData[v[0]],
945  m_vertexData[v[1]],
946  m_vertexData[v[2]]);
947  const Plane<T> trianglePlane(N, m_vertexData[v[0]]);
948  if (trianglePlane.isPointOnPositiveSide(m_vertexData[v[3]]))
949  {
950  std::swap(v[0], v[1]);
951  }
952  return m_mesh.setup(v[0], v[1], v[2], v[3]);
953  }
954 
955  // Find two most distant extreme points.
956  T maxD = m_epsilonSquared;
957  std::pair<size_t, size_t> selectedPoints;
958  for (size_t i = 0; i < 6; i++)
959  {
960  for (size_t j = i + 1; j < 6; j++)
961  {
962  const T d = m_vertexData[m_extremeValues[i]].getSquaredDistanceTo(m_vertexData[m_extremeValues[j]]);
963  if (d > maxD)
964  {
965  maxD = d;
966  selectedPoints = {m_extremeValues[i], m_extremeValues[j]};
967  }
968  }
969  }
970  if (maxD == m_epsilonSquared)
971  {
972  // A degenerate case: the point cloud seems to consists of a single point
973  return m_mesh.setup(0,
974  std::min((size_t)1, vertexCount - 1),
975  std::min((size_t)2, vertexCount - 1),
976  std::min((size_t)3, vertexCount - 1));
977  }
978  assert(selectedPoints.first != selectedPoints.second);
979 
980  // Find the most distant point to the line between the two chosen extreme points.
981  const Ray<T> r(m_vertexData[selectedPoints.first],
982  (m_vertexData[selectedPoints.second] - m_vertexData[selectedPoints.first]));
983  maxD = m_epsilonSquared;
984  size_t maxI = std::numeric_limits<size_t>::max();
985  const size_t vCount = m_vertexData.size();
986  for (size_t i = 0; i < vCount; i++)
987  {
988  const T distToRay = mathutils::getSquaredDistanceBetweenPointAndRay(m_vertexData[i], r);
989  if (distToRay > maxD)
990  {
991  maxD = distToRay;
992  maxI = i;
993  }
994  }
995  if (maxD == m_epsilonSquared)
996  {
997  // It appears that the point cloud belongs to a 1 dimensional
998  // subspace of R^3: convex hull has no volume => return a thin
999  // triangle. Pick any point other than selectedPoints.first and
1000  // selectedPoints.second as the third point of the triangle
1001 
1002  auto it = m_vertexData.begin();
1003  for (; it != m_vertexData.end() &&
1004  !(*it != m_vertexData[selectedPoints.first] && *it != m_vertexData[selectedPoints.second]);
1005  it++)
1006  {
1007  }
1008  const size_t thirdPoint = (it == m_vertexData.end()) ? selectedPoints.first : std::distance(m_vertexData.begin(), it);
1009 
1010  for (it = m_vertexData.begin();
1011  it != m_vertexData.end() &&
1012  !(*it != m_vertexData[selectedPoints.first] &&
1013  *it != m_vertexData[selectedPoints.second] &&
1014  *it != m_vertexData[thirdPoint]);
1015  it++)
1016  {
1017  }
1018  const size_t fourthPoint = (it == m_vertexData.end()) ? selectedPoints.first : std::distance(m_vertexData.begin(), it);
1019 
1020  return m_mesh.setup(selectedPoints.first, selectedPoints.second, thirdPoint, fourthPoint);
1021  }
1022 
1023  // These three points form the base triangle for our tetrahedron.
1024  assert(selectedPoints.first != maxI && selectedPoints.second != maxI);
1025  std::array<size_t, 3> baseTriangle{selectedPoints.first, selectedPoints.second, maxI};
1026  const Vector3<T> baseTriangleVertices[] = {m_vertexData[baseTriangle[0]],
1027  m_vertexData[baseTriangle[1]],
1028  m_vertexData[baseTriangle[2]]};
1029 
1030  // Next step is to find the 4th vertex of the tetrahedron. We naturally
1031  // choose the point farthest away from the triangle plane.
1032  maxD = m_epsilon;
1033  maxI = 0;
1034  const Vector3<T> N = mathutils::getTriangleNormal(baseTriangleVertices[0],
1035  baseTriangleVertices[1],
1036  baseTriangleVertices[2]);
1037  Plane<T> trianglePlane(N, baseTriangleVertices[0]);
1038  for (size_t i = 0; i < vCount; i++)
1039  {
1040  const T d = abs(mathutils::getSignedDistanceToPlane(m_vertexData[i], trianglePlane));
1041  if (d > maxD)
1042  {
1043  maxD = d;
1044  maxI = i;
1045  }
1046  }
1047  if (maxD == m_epsilon)
1048  {
1049  // All the points seem to lie on a 2D subspace of R^3. How to handle this? Well, let's add
1050  // one extra point to the point cloud so that the convex hull will have volume.
1051  m_planar = true;
1052  const vec3 N1 = mathutils::getTriangleNormal(baseTriangleVertices[1],
1053  baseTriangleVertices[2],
1054  baseTriangleVertices[0]);
1055  m_planarPointCloudTemp.clear();
1056  m_planarPointCloudTemp.insert(m_planarPointCloudTemp.begin(),
1057  m_vertexData.begin(),
1058  m_vertexData.end());
1059  const vec3 extraPoint = N1 + m_vertexData[0];
1060  m_planarPointCloudTemp.push_back(extraPoint);
1061  maxI = m_planarPointCloudTemp.size() - 1;
1062  m_vertexData = VertexDataSource<T>(m_planarPointCloudTemp);
1063  }
1064 
1065  // Enforce CCW orientation (if user prefers clockwise orientation,
1066  // swap two vertices in each triangle when final mesh is created)
1067  const Plane<T> triPlane(N, baseTriangleVertices[0]);
1068  if (triPlane.isPointOnPositiveSide(m_vertexData[maxI]))
1069  {
1070  std::swap(baseTriangle[0], baseTriangle[1]);
1071  }
1072 
1073  // Create a tetrahedron half edge mesh and compute planes defined by each triangle
1074  m_mesh.setup(baseTriangle[0], baseTriangle[1], baseTriangle[2], maxI);
1075  for (auto &f : m_mesh.m_faces)
1076  {
1077  auto v = m_mesh.getVertexIndicesOfFace(f);
1078  const Vector3<T> &va = m_vertexData[v[0]];
1079  const Vector3<T> &vb = m_vertexData[v[1]];
1080  const Vector3<T> &vc = m_vertexData[v[2]];
1081  const Vector3<T> N1 = mathutils::getTriangleNormal(va, vb, vc);
1082  const Plane<T> plane(N1, va);
1083  f.m_P = plane;
1084  }
1085 
1086  // Finally we assign a face for each vertex outside the tetrahedron
1087  // (vertices inside the tetrahedron have no role anymore)
1088  for (size_t i = 0; i < vCount; i++)
1089  {
1090  for (auto &face : m_mesh.m_faces)
1091  {
1092  if (addPointToFace(face, i))
1093  {
1094  break;
1095  }
1096  }
1097  }
1098  }

1101  // that they form a loop. Return true on success.
1102  inline bool reorderHorizonEdges(std::vector<size_t> &horizonEdges)
1103  {
1104  const size_t horizonEdgeCount = horizonEdges.size();
1105  for (size_t i = 0; i < horizonEdgeCount - 1; i++)
1106  {
1107  const size_t endVertex = m_mesh.m_halfEdges[horizonEdges[i]].m_endVertex;
1108  bool foundNext = false;
1109  for (size_t j = i + 1; j < horizonEdgeCount; j++)
1110  {
1111  const size_t beginVertex = m_mesh.m_halfEdges[m_mesh.m_halfEdges[horizonEdges[j]].m_opp].m_endVertex;
1112  if (beginVertex == endVertex)

1115  foundNext = true;
1116  break;
1117  }
1118  }
1119  if (!foundNext)
1120  {
1121  return false;
1122  }
1123  }
1124  assert(m_mesh.m_halfEdges[horizonEdges[horizonEdges.size() - 1]].m_endVertex == m_mesh.m_halfEdges[m_mesh.m_halfEdges[horizonEdges[0]].m_opp].m_endVertex);
1125  return true;

1222

1224  f[0].m_pointsOnPositiveSide->size() > 0)

1227  f[0].m_inFaceStack = 1;

1230  f[1].m_pointsOnPositiveSide->size() > 0)
1231  {
1232  m_faceList.push_back(1);
1233  f[1].m_inFaceStack = 1;

1236  f[2].m_pointsOnPositiveSide->size() > 0)
1237  {
1238  m_faceList.push_back(2);
1239  f[2].m_inFaceStack = 1;
1240  }
1241  if (f[3].m_pointsOnPositiveSide &&
1242  f[3].m_pointsOnPositiveSide->size() > 0)
1243  {
1244  m_faceList.push_back(3);
1245  f[3].m_inFaceStack = 1;
1246  }

1249  size_t iter = 0;
1250  while (!m_faceList.empty())
1251  {
1252  iter++;
1253  if (iter == std::numeric_limits<size_t>::max())
1254  {
1255  // Visible face traversal marks visited faces with iteration counter
1256  //(to mark that the face has been visited on this iteration) and the
1257  // max value represents unvisited faces. At this point we have to reset
1258  // iteration counter. This shouldn't be an issue on 64 bit machines.
1259  iter = 0;
1260  }
1261 
1262  const size_t topFaceIndex = m_faceList.front();
1263  m_faceList.pop_front();

1266  tf.m_inFaceStack = 0;
1267

1269  if (!tf.m_pointsOnPositiveSide || tf.isDisabled())
1270  {
1271  continue;
1272  }
1273

1275  // as the point to which we extrude
1276  const vec3 &activePoint = m_vertexData[tf.m_mostDistantPoint];
1277  const size_t activePointIndex = tf.m_mostDistantPoint;
1278 
1279  // Find out the faces that have our active point on their positive side
1280  // (these are the "visible faces"). The face on top of the stack of course
1281  // is one of them. At the same time, we create a list of horizon edges.
1282  m_horizonEdges.clear();
1283  m_possiblyVisibleFaces.clear();
1284  m_visibleFaces.clear();
1285  m_possiblyVisibleFaces.emplace_back(topFaceIndex, std::numeric_limits<size_t>::max());
1286  while (m_possiblyVisibleFaces.size())
1287  {
1288  const auto faceData = m_possiblyVisibleFaces.back();
1289  m_possiblyVisibleFaces.pop_back();

1291  assert(!pvf.isDisabled());

1295  if (pvf.m_isVisibleFaceOnCurrentIteration)
1296  {

1298  }
1299  }
1300  else
1301  {

1303  pvf.m_visibilityCheckedOnIteration = iter;
1304  const T d = P.m_N.dotProduct(activePoint) + P.m_D;
1305  if (d > 0)
1306  {
1307  pvf.m_isVisibleFaceOnCurrentIteration = 1;
1308  pvf.m_horizonEdgesOnCurrentIteration = 0;

1310  for (auto heIndex : m_mesh.getHalfEdgeIndicesOfFace(pvf))
1311  {
1312  if (m_mesh.m_halfEdges[heIndex].m_opp != faceData.m_enteredFromHalfEdge)
1313  {

1315  }

1323  // we came from is part of the horizon edge.
