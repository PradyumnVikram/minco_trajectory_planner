89  static inline void forwardT(const Eigen::VectorXd &tau,
90  Eigen::VectorXd &T)
91  {
92  const int sizeTau = tau.size();
93  T.resize(sizeTau);
94  for (int i = 0; i < sizeTau; i++)
95  {
96  T(i) = tau(i) > 0.0
97  ? ((0.5 * tau(i) + 1.0) * tau(i) + 1.0)
98  : 1.0 / ((0.5 * tau(i) - 1.0) * tau(i) + 1.0);
99  }
100  return;
101  }

103  template <typename EIGENVEC>
104  static inline void backwardT(const Eigen::VectorXd &T,
105  EIGENVEC &tau)
106  {
107  const int sizeT = T.size();
108  tau.resize(sizeT);
109  for (int i = 0; i < sizeT; i++)
110  {
111  tau(i) = T(i) > 1.0
112  ? (sqrt(2.0 * T(i) - 1.0) - 1.0)
113  : (1.0 - sqrt(2.0 / T(i) - 1.0));
114  }
115 
116  return;
117  }

119  template <typename EIGENVEC>
120  static inline void backwardGradT(const Eigen::VectorXd &tau,
121  const Eigen::VectorXd &gradT,
122  EIGENVEC &gradTau)
123  {
124  const int sizeTau = tau.size();
125  gradTau.resize(sizeTau);
126  double denSqrt;
127  for (int i = 0; i < sizeTau; i++)
128  {
129  if (tau(i) > 0)
130  {
131  gradTau(i) = gradT(i) * (tau(i) + 1.0);
132  }
133  else
134  {
135  denSqrt = (0.5 * tau(i) - 1.0) * tau(i) + 1.0;
136  gradTau(i) = gradT(i) * (1.0 - tau(i)) / (denSqrt * denSqrt);
137  }
138  }
139 
140  return;
141  }

143  static inline void forwardP(const Eigen::VectorXd &xi,
144  const Eigen::VectorXi &vIdx,
145  const PolyhedraV &vPolys,
146  Eigen::Matrix3Xd &P)
147  {
148  const int sizeP = vIdx.size();
149  P.resize(3, sizeP);
150  Eigen::VectorXd q;
151  for (int i = 0, j = 0, k, l; i < sizeP; i++, j += k)
152  {
153  l = vIdx(i);
154  k = vPolys[l].cols();
155  q = xi.segment(j, k).normalized().head(k - 1);
156  P.col(i) = vPolys[l].rightCols(k - 1) * q.cwiseProduct(q) +
157  vPolys[l].col(0);
158  }
159  return;
160  }

162  static inline double costTinyNLS(void *ptr,
163  const Eigen::VectorXd &xi,
164  Eigen::VectorXd &gradXi)
165  {
166  const int n = xi.size();
167  const Eigen::Matrix3Xd &ovPoly = *(Eigen::Matrix3Xd *)ptr;
168 
169  const double sqrNormXi = xi.squaredNorm();
170  const double invNormXi = 1.0 / sqrt(sqrNormXi);
171  const Eigen::VectorXd unitXi = xi * invNormXi;
172  const Eigen::VectorXd r = unitXi.head(n - 1);
173  const Eigen::Vector3d delta = ovPoly.rightCols(n - 1) * r.cwiseProduct(r) +
174  ovPoly.col(1) - ovPoly.col(0);
175 
176  double cost = delta.squaredNorm();
177  gradXi.head(n - 1) = (ovPoly.rightCols(n - 1).transpose() * (2 * delta)).array() *
178  r.array() * 2.0;
179  gradXi(n - 1) = 0.0;
180  gradXi = (gradXi - unitXi.dot(gradXi) * unitXi).eval() * invNormXi;
181 
182  const double sqrNormViolation = sqrNormXi - 1.0;
183  if (sqrNormViolation > 0.0)
184  {
185  double c = sqrNormViolation * sqrNormViolation;
186  const double dc = 3.0 * c;
187  c *= sqrNormViolation;
188  cost += c;
189  gradXi += dc * 2.0 * xi;
190  }
191 
192  return cost;
193  }

195  template <typename EIGENVEC>
196  static inline void backwardP(const Eigen::Matrix3Xd &P,
197  const Eigen::VectorXi &vIdx,
198  const PolyhedraV &vPolys,
199  EIGENVEC &xi)
200  {
201  const int sizeP = P.cols();
202 
203  double minSqrD;
204  lbfgs::lbfgs_parameter_t tiny_nls_params;
205  tiny_nls_params.past = 0;
206  tiny_nls_params.delta = 1.0e-5;
207  tiny_nls_params.g_epsilon = FLT_EPSILON;
208  tiny_nls_params.max_iterations = 128;
209 
210  Eigen::Matrix3Xd ovPoly;
211  for (int i = 0, j = 0, k, l; i < sizeP; i++, j += k)
212  {
213  l = vIdx(i);
214  k = vPolys[l].cols();
215 
216  ovPoly.resize(3, k + 1);
217  ovPoly.col(0) = P.col(i);
218  ovPoly.rightCols(k) = vPolys[l];
219  Eigen::VectorXd x(k);
220  x.setConstant(sqrt(1.0 / k));
221  lbfgs::lbfgs_optimize(x,
222  minSqrD,
223  &GCOPTER_PolytopeSFC::costTinyNLS,
224  nullptr,
225  nullptr,
226  &ovPoly,
227  tiny_nls_params);
228 
229  xi.segment(j, k) = x;
230  }
231 
232  return;
233  }

235  template <typename EIGENVEC>
236  static inline void backwardGradP(const Eigen::VectorXd &xi,
237  const Eigen::VectorXi &vIdx,
238  const PolyhedraV &vPolys,
239  const Eigen::Matrix3Xd &gradP,
240  EIGENVEC &gradXi)
241  {
242  const int sizeP = vIdx.size();
243  gradXi.resize(xi.size());
244 
245  double normInv;
246  Eigen::VectorXd q, gradQ, unitQ;
247  for (int i = 0, j = 0, k, l; i < sizeP; i++, j += k)
248  {
249  l = vIdx(i);
250  k = vPolys[l].cols();
251  q = xi.segment(j, k);
252  normInv = 1.0 / q.norm();
253  unitQ = q * normInv;
254  gradQ.resize(k);
255  gradQ.head(k - 1) = (vPolys[l].rightCols(k - 1).transpose() * gradP.col(i)).array() *
256  unitQ.head(k - 1).array() * 2.0;
257  gradQ(k - 1) = 0.0;
258  gradXi.segment(j, k) = (gradQ - unitQ * unitQ.dot(gradQ)) * normInv;
259  }
260 
261  return;
262  }

264  template <typename EIGENVEC>
265  static inline void normRetrictionLayer(const Eigen::VectorXd &xi,
266  const Eigen::VectorXi &vIdx,
267  const PolyhedraV &vPolys,
268  double &cost,
269  EIGENVEC &gradXi)
270  {
271  const int sizeP = vIdx.size();
272  gradXi.resize(xi.size());
273 
274  double sqrNormQ, sqrNormViolation, c, dc;
275  Eigen::VectorXd q;
276  for (int i = 0, j = 0, k; i < sizeP; i++, j += k)
277  {
278  k = vPolys[vIdx(i)].cols();
279 
280  q = xi.segment(j, k);
281  sqrNormQ = q.squaredNorm();
282  sqrNormViolation = sqrNormQ - 1.0;
283  if (sqrNormViolation > 0.0)
284  {
285  c = sqrNormViolation * sqrNormViolation;
286  dc = 3.0 * c;
287  c *= sqrNormViolation;
288  cost += c;
289  gradXi.segment(j, k) += dc * 2.0 * q;
290  }
291  }
292 
293  return;
294  }

296  static inline bool smoothedL1(const double &x,
297  const double &mu,
298  double &f,
299  double &df)
300  {
301  if (x < 0.0)
302  {
303  return false;
304  }
305  else if (x > mu)
306  {
307  f = x - 0.5 * mu;
308  df = 1.0;
309  return true;
310  }
311  else
312  {
313  const double xdmu = x / mu;
314  const double sqrxdmu = xdmu * xdmu;
315  const double mumxd2 = mu - 0.5 * x;
316  f = mumxd2 * sqrxdmu * xdmu;
317  df = sqrxdmu * ((-0.5) * xdmu + 3.0 * mumxd2 / mu);
318  return true;
319  }
320  }

322  // magnitudeBounds = [v_max, omg_max, theta_max, thrust_min, thrust_max]^T
323  // penaltyWeights = [pos_weight, vel_weight, omg_weight, theta_weight, thrust_weight]^T
324  // physicalParams = [vehicle_mass, gravitational_acceleration, horitonral_drag_coeff,
325  // vertical_drag_coeff, parasitic_drag_coeff, speed_smooth_factor]^T
326  static inline void attachPenaltyFunctional(const Eigen::VectorXd &T,
327  const Eigen::MatrixX3d &coeffs,
328  const Eigen::VectorXi &hIdx,
329  const PolyhedraH &hPolys,
330  const double &smoothFactor,
331  const int &integralResolution,
332  const Eigen::VectorXd &magnitudeBounds,
333  const Eigen::VectorXd &penaltyWeights,
334  flatness::FlatnessMap &flatMap,
335  double &cost,
336  Eigen::VectorXd &gradT,
337  Eigen::MatrixX3d &gradC)
338  {
339  const double velSqrMax = magnitudeBounds(0) * magnitudeBounds(0);
340  const double omgSqrMax = magnitudeBounds(1) * magnitudeBounds(1);
341  const double thetaMax = magnitudeBounds(2);
342  const double thrustMean = 0.5 * (magnitudeBounds(3) + magnitudeBounds(4));
343  const double thrustRadi = 0.5 * fabs(magnitudeBounds(4) - magnitudeBounds(3));
344  const double thrustSqrRadi = thrustRadi * thrustRadi;
345 
346  const double weightPos = penaltyWeights(0);
347  const double weightVel = penaltyWeights(1);
348  const double weightOmg = penaltyWeights(2);
349  const double weightTheta = penaltyWeights(3);
350  const double weightThrust = penaltyWeights(4);
351 
352  Eigen::Vector3d pos, vel, acc, jer, sna;
353  Eigen::Vector3d totalGradPos, totalGradVel, totalGradAcc, totalGradJer;
354  double totalGradPsi, totalGradPsiD;
355  double thr, cos_theta;
356  Eigen::Vector4d quat;
357  Eigen::Vector3d omg;
358  double gradThr;
359  Eigen::Vector4d gradQuat;
360  Eigen::Vector3d gradPos, gradVel, gradOmg;
361 
362  double step, alpha;
363  double s1, s2, s3, s4, s5;
364  Eigen::Matrix<double, 6, 1> beta0, beta1, beta2, beta3, beta4;
365  Eigen::Vector3d outerNormal;
366  int K, L;
367  double violaPos, violaVel, violaOmg, violaTheta, violaThrust;
368  double violaPosPenaD, violaVelPenaD, violaOmgPenaD, violaThetaPenaD, violaThrustPenaD;
369  double violaPosPena, violaVelPena, violaOmgPena, violaThetaPena, violaThrustPena;
370  double node, pena;
371 
372  const int pieceNum = T.size();
373  const double integralFrac = 1.0 / integralResolution;
374  for (int i = 0; i < pieceNum; i++)
375  {
376  const Eigen::Matrix<double, 6, 3> &c = coeffs.block<6, 3>(i * 6, 0);
377  step = T(i) * integralFrac;
378  for (int j = 0; j <= integralResolution; j++)
379  {
380  s1 = j * step;
381  s2 = s1 * s1;
382  s3 = s2 * s1;
383  s4 = s2 * s2;
384  s5 = s4 * s1;
385  beta0(0) = 1.0, beta0(1) = s1, beta0(2) = s2, beta0(3) = s3, beta0(4) = s4, beta0(5) = s5;
386  beta1(0) = 0.0, beta1(1) = 1.0, beta1(2) = 2.0 * s1, beta1(3) = 3.0 * s2, beta1(4) = 4.0 * s3, beta1(5) = 5.0 * s4;
387  beta2(0) = 0.0, beta2(1) = 0.0, beta2(2) = 2.0, beta2(3) = 6.0 * s1, beta2(4) = 12.0 * s2, beta2(5) = 20.0 * s3;
388  beta3(0) = 0.0, beta3(1) = 0.0, beta3(2) = 0.0, beta3(3) = 6.0, beta3(4) = 24.0 * s1, beta3(5) = 60.0 * s2;
389  beta4(0) = 0.0, beta4(1) = 0.0, beta4(2) = 0.0, beta4(3) = 0.0, beta4(4) = 24.0, beta4(5) = 120.0 * s1;
390  pos = c.transpose() * beta0;
391  vel = c.transpose() * beta1;
392  acc = c.transpose() * beta2;
393  jer = c.transpose() * beta3;
394  sna = c.transpose() * beta4;
395 
396  flatMap.forward(vel, acc, jer, 0.0, 0.0, thr, quat, omg);
397 
398  violaVel = vel.squaredNorm() - velSqrMax;
399  violaOmg = omg.squaredNorm() - omgSqrMax;
400  cos_theta = 1.0 - 2.0 * (quat(1) * quat(1) + quat(2) * quat(2));
401  violaTheta = acos(cos_theta) - thetaMax;
402  violaThrust = (thr - thrustMean) * (thr - thrustMean) - thrustSqrRadi;
403 
404  gradThr = 0.0;
405  gradQuat.setZero();
406  gradPos.setZero(), gradVel.setZero(), gradOmg.setZero();
407  pena = 0.0;
408 
409  L = hIdx(i);
410  K = hPolys[L].rows();
411  for (int k = 0; k < K; k++)
412  {
413  outerNormal = hPolys[L].block<1, 3>(k, 0);
414  violaPos = outerNormal.dot(pos) + hPolys[L](k, 3);
415  if (smoothedL1(violaPos, smoothFactor, violaPosPena, violaPosPenaD))
416  {
417  gradPos += weightPos * violaPosPenaD * outerNormal;
418  pena += weightPos * violaPosPena;
419  }
420  }
421 
422  if (smoothedL1(violaVel, smoothFactor, violaVelPena, violaVelPenaD))
423  {
424  gradVel += weightVel * violaVelPenaD * 2.0 * vel;
425  pena += weightVel * violaVelPena;
426  }
427 
428  if (smoothedL1(violaOmg, smoothFactor, violaOmgPena, violaOmgPenaD))
429  {
430  gradOmg += weightOmg * violaOmgPenaD * 2.0 * omg;
431  pena += weightOmg * violaOmgPena;
432  }
433 
434  if (smoothedL1(violaTheta, smoothFactor, violaThetaPena, violaThetaPenaD))
435  {
436  gradQuat += weightTheta * violaThetaPenaD /
437  sqrt(1.0 - cos_theta * cos_theta) * 4.0 *
438  Eigen::Vector4d(0.0, quat(1), quat(2), 0.0);
439  pena += weightTheta * violaThetaPena;
440  }
441 
442  if (smoothedL1(violaThrust, smoothFactor, violaThrustPena, violaThrustPenaD))
443  {
444  gradThr += weightThrust * violaThrustPenaD * 2.0 * (thr - thrustMean);
445  pena += weightThrust * violaThrustPena;
446  }
447 
448  flatMap.backward(gradPos, gradVel, gradThr, gradQuat, gradOmg,
449  totalGradPos, totalGradVel, totalGradAcc, totalGradJer,
450  totalGradPsi, totalGradPsiD);
451 
452  node = (j == 0 || j == integralResolution) ? 0.5 : 1.0;
453  alpha = j * integralFrac;
454  gradC.block<6, 3>(i * 6, 0) += (beta0 * totalGradPos.transpose() +
455  beta1 * totalGradVel.transpose() +
456  beta2 * totalGradAcc.transpose() +
457  beta3 * totalGradJer.transpose()) *
458  node * step;
459  gradT(i) += (totalGradPos.dot(vel) +
460  totalGradVel.dot(acc) +
461  totalGradAcc.dot(jer) +
462  totalGradJer.dot(sna)) *
463  alpha * node * step +
464  node * integralFrac * pena;
465  cost += node * step * pena;
466  }
467  }
468 
469  return;
470  }

472  static inline double costFunctional(void *ptr,
473  const Eigen::VectorXd &x,
474  Eigen::VectorXd &g)
475  {
476  GCOPTER_PolytopeSFC &obj = *(GCOPTER_PolytopeSFC *)ptr;
477  const int dimTau = obj.temporalDim;
478  const int dimXi = obj.spatialDim;
479  const double weightT = obj.rho;
480  Eigen::Map<const Eigen::VectorXd> tau(x.data(), dimTau);
481  Eigen::Map<const Eigen::VectorXd> xi(x.data() + dimTau, dimXi);
482  Eigen::Map<Eigen::VectorXd> gradTau(g.data(), dimTau);
483  Eigen::Map<Eigen::VectorXd> gradXi(g.data() + dimTau, dimXi);
484 
485  forwardT(tau, obj.times);
486  forwardP(xi, obj.vPolyIdx, obj.vPolytopes, obj.points);
487 
488  double cost;
489  obj.minco.setParameters(obj.points, obj.times);
490  obj.minco.getEnergy(cost);
491  obj.minco.getEnergyPartialGradByCoeffs(obj.partialGradByCoeffs);
492  obj.minco.getEnergyPartialGradByTimes(obj.partialGradByTimes);
493 
494  attachPenaltyFunctional(obj.times, obj.minco.getCoeffs(),
495  obj.hPolyIdx, obj.hPolytopes,
496  obj.smoothEps, obj.integralRes,
497  obj.magnitudeBd, obj.penaltyWt, obj.flatmap,
498  cost, obj.partialGradByTimes, obj.partialGradByCoeffs);
499 
500  obj.minco.propogateGrad(obj.partialGradByCoeffs, obj.partialGradByTimes,
501  obj.gradByPoints, obj.gradByTimes);
502 
503  cost += weightT * obj.times.sum();
504  obj.gradByTimes.array() += weightT;
505 
506  backwardGradT(tau, obj.gradByTimes, gradTau);
507  backwardGradP(xi, obj.vPolyIdx, obj.vPolytopes, obj.gradByPoints, gradXi);
508  normRetrictionLayer(xi, obj.vPolyIdx, obj.vPolytopes, cost, gradXi);
509 
510  return cost;
511  }

513  static inline double costDistance(void *ptr,
514  const Eigen::VectorXd &xi,
515  Eigen::VectorXd &gradXi)
516  {
517  void **dataPtrs = (void **)ptr;
518  const double &dEps = *((const double *)(dataPtrs[0]));
519  const Eigen::Vector3d &ini = *((const Eigen::Vector3d *)(dataPtrs[1]));
520  const Eigen::Vector3d &fin = *((const Eigen::Vector3d *)(dataPtrs[2]));
521  const PolyhedraV &vPolys = *((PolyhedraV *)(dataPtrs[3]));
522 
523  double cost = 0.0;
524  const int overlaps = vPolys.size() / 2;
525 
526  Eigen::Matrix3Xd gradP = Eigen::Matrix3Xd::Zero(3, overlaps);
527  Eigen::Vector3d a, b, d;
528  Eigen::VectorXd r;
529  double smoothedDistance;
530  for (int i = 0, j = 0, k = 0; i <= overlaps; i++, j += k)
531  {
532  a = i == 0 ? ini : b;
533  if (i < overlaps)
534  {
535  k = vPolys[2 * i + 1].cols();
536  Eigen::Map<const Eigen::VectorXd> q(xi.data() + j, k);
537  r = q.normalized().head(k - 1);
538  b = vPolys[2 * i + 1].rightCols(k - 1) * r.cwiseProduct(r) +
539  vPolys[2 * i + 1].col(0);
540  }
541  else
542  {
543  b = fin;
544  }
545 
546  d = b - a;
547  smoothedDistance = sqrt(d.squaredNorm() + dEps);
548  cost += smoothedDistance;
549 
550  if (i < overlaps)
551  {
552  gradP.col(i) += d / smoothedDistance;
553  }
554  if (i > 0)
555  {
556  gradP.col(i - 1) -= d / smoothedDistance;
557  }
558  }
559 
560  Eigen::VectorXd unitQ;
561  double sqrNormQ, invNormQ, sqrNormViolation, c, dc;
562  for (int i = 0, j = 0, k; i < overlaps; i++, j += k)
563  {
564  k = vPolys[2 * i + 1].cols();
565  Eigen::Map<const Eigen::VectorXd> q(xi.data() + j, k);
566  Eigen::Map<Eigen::VectorXd> gradQ(gradXi.data() + j, k);
567  sqrNormQ = q.squaredNorm();
568  invNormQ = 1.0 / sqrt(sqrNormQ);
569  unitQ = q * invNormQ;
570  gradQ.head(k - 1) = (vPolys[2 * i + 1].rightCols(k - 1).transpose() * gradP.col(i)).array() *
571  unitQ.head(k - 1).array() * 2.0;
572  gradQ(k - 1) = 0.0;
573  gradQ = (gradQ - unitQ * unitQ.dot(gradQ)).eval() * invNormQ;
574 
575  sqrNormViolation = sqrNormQ - 1.0;
576  if (sqrNormViolation > 0.0)
577  {
578  c = sqrNormViolation * sqrNormViolation;
579  dc = 3.0 * c;
580  c *= sqrNormViolation;
581  cost += c;
582  gradQ += dc * 2.0 * q;
583  }
584  }
585 
586  return cost;
587  }
